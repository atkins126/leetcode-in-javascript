# [24] Swap Nodes in Pairs

```js
var swapPairs = function(head) {
  const dude = new ListNode(null);
  dude.next = head;
  let pre = dude;
  while (pre && pre.next && pre.next.next) {
    let cur = pre.next;
    let next = cur.next;
    pre.next = next;
    cur.next = next.next;
    next.next = cur;

    pre = cur;
  }
  return dude.next;
};

function ListNode(val) {
  this.val = val;
  this.next = null;
}
```
