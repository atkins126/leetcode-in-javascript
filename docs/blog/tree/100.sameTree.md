# [100] Same Tree

```js
var isSameTree = function(p, q) {
  if (!p && !q) return true;
  if ((!p && q) || (p && !q) || p.val !== q.val) {
    return false;
  }
  const l = isSameTree(p.left, q.left);
  const r = isSameTree(p.right, q.right);
  return l && r;
};

function TreeNode(val) {
  this.val = val;
  this.left = this.right = null;
}
```
