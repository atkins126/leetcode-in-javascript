# [7] Reverse Integer

```js
var reverse = function(x) {
  const INT_MAX = Math.pow(2, 31) - 1;
  const INT_MIN = Math.pow(-2, 31);
  let res = Math.abs(x)
    .toString()
    .split('')
    .reverse();

  while (res[0] === '0') {
    res.shift();
  }

  if (x < 0) {
    res.unshift('-');
  }
  res = res.join('');
  if (res < INT_MIN || res > INT_MAX) {
    return 0;
  }
  return res;
};
```
