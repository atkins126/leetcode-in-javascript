# [77] 组合

给定两个整数 n 和 k，返回 1 ... n 中所有可能的 k 个数的组合。

示例:

输入: n = 4, k = 2

输出:

[

⁠ [2,4],

⁠ [3,4],

⁠ [2,3],

⁠ [1,2],

⁠ [1,3],

⁠ [1,4],

]

var combine = function(n, k) {
  const res = [];
  backtrack(k, [], 1);
  return res;

  function backtrack(k, path, start) {
    if (k === 0) {
      res.push([...path]);
      return;
    }
    for (let i = start; i <= n; i++) {
      path.push(i);
      backtrack(k - 1, path, i + 1);
      path.pop();
    }
  }
};
